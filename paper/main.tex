\begin{lemma}
    Let $X=\{\x_1,...,\x_n\}$ be a set of vectors in $\R^d$ and $\sigma$ be the RELU activation function.
    Then,
    \begin{equation}
        \ltwo{\x_i-\x_j} \ge \ltwo{\sigma(\x_i)-\sigma(\x_j)}
    \end{equation}
    for all $i,j$ in $\{1,...,n\}$.
\end{lemma}
